module list where

    postulate A : Set
    postulate a : A
    postulate b : A
    postulate c : A

    infixr 40 _::_

    data  List  (A : Set ) : Set  where
       [] : List A
       _::_ : A → List A → List A

    infixl 30 _++_

    _++_ :   {A : Set } → List A → List A → List A
    []        ++ ys = ys
    (x :: xs) ++ ys = x :: (xs ++ ys)

    l1 = a :: []
    l2 = a :: b :: a :: c ::  []
    l3 = l1 ++ l2
    l4 = l3 ++ l3

    --
    --    ::  →  :: → :: →  []
    --    |      |    |
    --    a      b    c
